77  int inserted=0;
78 
79  level=ROOT_LEVEL-1;
80  while(!inserted) {
81  if(octree[octIdx].data==-1) { /* insert cell into node */
82  octree[octIdx].data=c;

86  childBranchBit = 1 << (level);

88  ybit=(locCode[c].y) & childBranchBit;
89  zbit=(locCode[c].z) & childBranchBit;
90  childIndex = ( ((xbit) >> (level))
91  + ((ybit) >> (level-1))
92  + ((zbit) >> (level-2)) );
93  level--;
94  father=octIdx;

96  if(child==-1) {
97  octree[octIdx].child[childIndex]=octSize;
98  child=octSize;

100  }

102  } else { /* occupied leaf */

105  childBranchBit = 1 << (level);
106  xbit=(locCode[oc].x) & childBranchBit;
107  ybit=(locCode[oc].y) & childBranchBit;
108  zbit=(locCode[oc].z) & childBranchBit;

129 void octBuild()
130 {
131  int i,c;
132  int64_t octMaxSize;
133  struct doubleVector3d bMin,bMax;
134  double epsilon=0.01;
135 
136  if(lnc==0)
137  return;
138 
139  bMin.x= DBL_MAX;
140  bMin.y= DBL_MAX;
141  bMin.z= DBL_MAX;
142  bMax.x=-DBL_MAX;
143  bMax.y=-DBL_MAX;
144  bMax.z=-DBL_MAX;
145  for(i=0; i<lnc; i++) {
146  double x,y,z;
147  double e;
148  x=cells[i].x;
149  y=cells[i].y;
150  z=cells[i].z;
151  e=h+epsilon;
152  bMin.x=(x-e<bMin.x?x-e:bMin.x);
153  bMax.x=(x+e>bMax.x?x+e:bMax.x);
154  bMin.y=(y-e<bMin.y?y-e:bMin.y);
155  bMax.y=(y+e>bMax.y?y+e:bMax.y);
156  bMin.z=(z-e<bMin.z?z-e:bMin.z);
157  bMax.z=(z+e>bMax.z?z+e:bMax.z);
158  }
159  affScale=bMax.x-bMin.x;
160  affScale=(affScale>bMax.y-bMin.y?affScale:bMax.y-bMin.y);
161  affScale=(affScale>bMax.z-bMin.z?affScale:bMax.z-bMin.z);
162  affShift.x=bMin.x;
163  affShift.y=bMin.y;
164  affShift.z=bMin.z;
165  /* each cell coordinate will be shifted by affShift and scaled by affScale */
166 
167  locCode=(struct uintVector3d*) malloc(sizeof(struct uintVector3d)*lnc);
168  //#pragma omp parallel for
169  for(c=0; c<lnc; c++) {
170  locCode[c].x=(unsigned int)( ((cells[c].x-affShift.x)/affScale) * MAXVALUE );
171  locCode[c].y=(unsigned int)( ((cells[c].y-affShift.y)/affScale) * MAXVALUE );
172  locCode[c].z=(unsigned int)( ((cells[c].z-affShift.z)/affScale) * MAXVALUE );
173  }
174  /* memory space required to store octree (to be reviewed again!) */

180 #endif
181  root=0;
182  octSize=0;
183  octEmptyNode(-1,ROOT_LEVEL,0,0,0);
184 
185  for(c=0; c<lnc; c++) {

258 MIC_ATTR int octTraverseToLevel(unsigned int level,unsigned int xloc,unsigned int yloc,unsigned int zloc,unsigned int lmin)
259 {
260  int cellIdx=0;
261  int parent=0;
262  unsigned int n;
263  unsigned int childBranchBit,childIndex;
264  n=(level)-(lmin)+1;
265  while(n--) {
266  childBranchBit = 1 << (level);
267  childIndex = ( (((xloc) & childBranchBit) >> (level))
268  + (((yloc) & childBranchBit) >> (level-1))
269  + (((zloc) & childBranchBit) >> ((level-2))) );
270  level--;
271  parent=cellIdx;
272  cellIdx=octree[cellIdx].child[childIndex];
273  if(octree[cellIdx].data != -2) break; /* a leaf */
274  }
275  if(cellIdx==-1) cellIdx=parent;
276  return cellIdx;

282 MIC_ATTR int octLocateRegion(struct uintVector3d minLocCode,struct uintVector3d maxLocCode)
283 {
284  unsigned int l1,l2,lmin;
285  unsigned int level;
286  int cellIdx;
287  struct uintVector3d locDiff;
288  /* XOR of location codes */
289  locDiff.x=minLocCode.x ^ maxLocCode.x;
290  locDiff.y=minLocCode.y ^ maxLocCode.y;
291  locDiff.z=minLocCode.z ^ maxLocCode.z;
292  /* Determining the level of the smallest cell containing the region */
293  l1=ROOT_LEVEL;
294  l2=ROOT_LEVEL;
295  lmin=ROOT_LEVEL;
296  while(!(locDiff.x & (1<<l1)) && l1) l1--;
297  while(!(locDiff.y & (1<<l2)) && (l2>l1)) l2--;
298  while(!(locDiff.z & (1<<lmin)) && (lmin>l2)) lmin--;
299  lmin++;
300  level=ROOT_LEVEL-1;
301  cellIdx=octTraverseToLevel(level,minLocCode.x,minLocCode.y,minLocCode.z,lmin);
302  return cellIdx;
303 }
